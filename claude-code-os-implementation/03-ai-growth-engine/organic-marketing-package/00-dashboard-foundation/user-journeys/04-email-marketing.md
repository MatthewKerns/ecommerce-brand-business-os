interface EmailPlatformConnection {

interface EmailSequence {
  id: string;
  workspaceId: string;
  platformConnectionId: string;
  externalId?: string; // ID in the connected platform

  type: 'welcome' | 'nurture' | 'post_purchase' | 'win_back' | 'custom';
